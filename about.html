const floor3 = (x) => {
	const x2 = x >> 1;
	const x3 = x >> 2;
	const b1 = (x >> 3) | (x2 & x3);
	const b2 = (~x2 | x) & (x2 ^ x3);
	return (b1 << 1) | b2;
}

/* mod x%3

b1 1xxx | ((x2 ^ x1) & (x2 ^ x3))
b2 x111 | (~x2 & (x1 ^ x3))

1000 10
0101 10
0010 10

0111 01
0100 01
0001 01

0110 00
0011 00
0000 00
*/

const mod3 = (x) => {
	const x1 = x & 0x1;
	const x2 = (x >> 1) & 0x1;
	const x3 = x >> 2;

	const b1 = (x >> 3) | ((x1 ^ x2) & (x2 ^ x3));
	const b2 = ~x2 & (x1 ^ x3);
	return (b1 << 1) | (b2 | (x1 & x2 & x3));
}
