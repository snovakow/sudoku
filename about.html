const floor3 = (x) => {
	const x2 = x >> 1;
	const x3 = x >> 2;
	const b1 = (x >> 3) | (x2 & x3);
	const b2 = (~x2 | x) & (x2 ^ x3);
	return (b1 << 1) | b2;
}
